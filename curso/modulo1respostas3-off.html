9) Solicite duas frações a/b e c/d, e imprima a fração e/f resultante do produto entre as duas frações . 

int a, b; // primeira fracao a/b
int c, d; // segunda fracao c/d
int e, f; // fracao resultado e/f

// obtendo primeira fracao
Console.Write("Digite o valor A, numerador da primeira fração: ");
a = Int32.Parse(Console.ReadLine());
Console.Write("Digite o valor B, denominador da primeira fração: ");
b = Int32.Parse(Console.ReadLine());

// validando primeira fração, denominador não pode ser igual a 0
if (b != 0)
{
    // O USUARIO DIGITOU A PRIMEIRA FRACAO VALIDA!

    // obtendo a segunda fracao
    Console.Write("Digite o valor C, numerador da segunda fração: ");
    c = Int32.Parse(Console.ReadLine());
    Console.Write("Digite o valor D, denominador da segunda fração: ");
    d = Int32.Parse(Console.ReadLine());

    // validando segunda fração, denominador não pode ser igual a 0
    if (d != 0)
    {
        // O USUARIO DIGITOU A SEGUNDA FRACAO VALIDA!

        // multiplicando as duas frações
        e = a * c; // numerador
        f = b * d; // denominador

        Console.WriteLine("A multiplicação entre {0}/{1} e {2}/{3} = {4}/{5}.", a, b, c, d, e, f);
    }
    else // segunda fração inválida!
    {
        Console.WriteLine("ERRO! Fração {0}/{1} inválida!!", c, d);
    }
}
else // primeira fração inválida!
{
    Console.WriteLine("ERRO! Fração {0}/{1} inválida!!", a, b);
}

12) Modifique o programa acima para utilizar uma variável do tipo enum ao invés de um char. 

enum Operacao
{
    soma,
    subtracao,
    multiplicacao,
    divisao
}

class Program
{
    static void Main(string[] args)
    {
        double a, b;
        double resultado;
        Operacao operacao;

        Console.Write("Digite o valor de A: ");
        a = Double.Parse(Console.ReadLine());

        Console.Write("Digite o valor de B: ");
        b = Double.Parse(Console.ReadLine());

        Console.Write("Digite a operação: ");
        operacao = (Operacao)Enum.Parse(typeof(Operacao), Console.ReadLine());

        switch (operacao)
        {
            case Operacao.soma:
                resultado = a + b;
                Console.WriteLine("{0} + {1} = {2}", a, b, resultado);
                break;
            case Operacao.subtracao:
                resultado = a - b;
                Console.WriteLine("{0} - {1} = {2}", a, b, resultado);
                break;
            case Operacao.multiplicacao:
                resultado = a * b;
                Console.WriteLine("{0} * {1} = {2}", a, b, resultado);
                break;
            case Operacao.divisao:
                resultado = a / b;
                Console.WriteLine("{0} / {1} = {2}", a, b, resultado);
                break;
            default:
                Console.WriteLine("Operação {0} inválida!", operacao);
                break;
        }

        Console.ReadKey();
    }
}
