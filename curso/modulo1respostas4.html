5) Dado N , calcular a soma dos N primeiros números naturais. 

int n;
int soma;

// solicitando ao usuário o valor de n
Console.Write("Digite o valor de N: ");
n = Int32.Parse(Console.ReadLine());

// Exemplos:
// - Para N = 1: 0                 = 0
// - Para N = 2: 0 + 1             = 1
// - Para N = 3: 0 + 1 + 2         = 3
// - Para N = 4: 0 + 1 + 2 + 3     = 6
// - Para N = 5: 0 + 1 + 2 + 3 + 4 = 10

soma = 0;

for (int i = 0; i < n; i++)
    soma += i;

Console.WriteLine("A soma dos {0} primeiros números naturais é igual a {1}", n, soma);
Console.ReadKey();

9) Uma loja de discos anota diariamente durante o mês de março a quantidade de discos vendidos. Determinar em que dia desse mês ocorreu a maior venda e qual foi a quantidade de discos vendida nesse dia. 

int venda;
int maiorvenda;
int diamaiorvenda;

// solicitando ao usuário o valor da venda do primeiro dia
Console.Write("Digite a quantidade de discos vendidos no dia 1: ");
venda = Int32.Parse(Console.ReadLine());

maiorvenda = venda;
diamaiorvenda = 1;

for (int i = 2; i <= 31; i++)
{
    // solicitando ao usuário o valor da venda do primeiro dia
    Console.Write("Digite a quantidade de discos vendidos no dia {0}: ", i);
    venda = Int32.Parse(Console.ReadLine());

    if (venda > maiorvenda)
    {
        maiorvenda = venda;
        diamaiorvenda = i;
    }
}

Console.WriteLine("A maior venda ocorreu no dia {0}, e o número de discos vendidos nesse dia foi {1}.", diamaiorvenda, maiorvenda);
Console.ReadKey();
