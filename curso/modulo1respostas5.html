1) Dada uma sequência de N números, imprimi-la na ordem inversa. 

int[] sequencia; // vetor de inteiros
int n; // tamanho do vetor de inteiros (número de elementos)

// solicitando o valor de N
Console.Write("Quantos elementos a sequência precisa ter? ");
n = Int32.Parse(Console.ReadLine());

// alocando N elementos para o vetor de inteiros
sequencia = new int[n];

// solicitar ao usuário digitar o valor de todos os elementos do vetor
// estamos percorrendo os N elementos, para o usuário digitar o valor de cada elemento
for (int i = 0; i < n; i++)
{
    // solicitando ao usuário digitar o valor para armazenar na posição i
    Console.Write("Digite o valor para ser armazenado na posição {0}: ", i+1);
    sequencia[i] = Int32.Parse(Console.ReadLine());
}
Console.WriteLine();

// imprimindo todos os elementos do vetor na ordem inversa
for (int i = n-1; i >= 0; i--)
    Console.WriteLine("Posição {0}, valor {1}", i+1, sequencia[i]);

Console.ReadKey();

2) Dado um número N, imprimi-lo como uma soma de múltiplos de 10 mais a sua unidade.
Exemplo: 472 = 400 + 70 + 2 

int numero; // numero que o usuário vai digitar
int[] casasdecimais; // numero representado em casas decimais
int divisor, resto;
int numcasasdecimais;
int multiplicador;
int i;

// solicitando ao usuário para digitar o número
Console.Write("Digite um número: ");
numero = Int32.Parse(Console.ReadLine());

// descobrindo qual é o número de casas decimais do número digitado pelo usuário
divisor = numero;
numcasasdecimais = 0;
while (divisor > 0)
{
    resto = divisor % 10;
    divisor /= 10;
    numcasasdecimais++;
}

// alocando o vetor com o tamanho igual ao número de casas decimais
casasdecimais = new int[numcasasdecimais];

// calculando multiplicador
multiplicador = (int) Math.Pow(10, numcasasdecimais-1);

Console.Write("{0} = ", numero);

divisor = numero;
i = 0;
while (multiplicador > 0)
{
    if (multiplicador == 1)
        Console.Write("{0}", (divisor / multiplicador) * multiplicador);
    else
        Console.Write("{0} + ", (divisor / multiplicador) * multiplicador);

    // armazenando casa decimal atual na posicao i do vetor
    casasdecimais[i] = divisor / multiplicador;

    divisor %= multiplicador;
    multiplicador /= 10;
    i++;
}
Console.WriteLine();

// imprimindo os elementos do vetor
Console.WriteLine();
for (i = 0; i < numcasasdecimais; i++)
    System.Console.WriteLine("Posição {0}, valor {1}", i, casasdecimais[i]);

Console.ReadKey();

3) Números palíndromos são aqueles que são lidos da direita para a esquerda da mesma maneira que da esquerda para a direita.
Exemplo: o número 12321 é palíndromo, enquanto 123 não é.
Dado um número N, informar se ele é palíndromo ou não. 

int numero; // variável que o usuário digita
int[] casasdecimais; // numero representado em casas decimais
int[] inverso; // inverso do número que o usuário digita
int numcasasdecimais; // numero de casas decimais do número digitado pelo usuário
int divisor;
int resto;
int multiplicador;
int i;
int j;
bool palindromo;

Console.Write("Digite um número: ");

// descobrindo qual é o número de casas decimais do número digitado pelo usuário
divisor = numero;
numcasasdecimais = 0;
while (divisor > 0)
{
    resto = divisor % 10;
    divisor /= 10;
    numcasasdecimais++;
}
// nesse ponto o programa sabe o número de casas decimais

// alocando o vetor com o tamanho igual ao número de casas decimais
casasdecimais = new int[numcasasdecimais];

// calculando multiplicador
multiplicador = (int) Math.Pow(10, numcasasdecimais-1);

divisor = numero;
i = 0;
while (multiplicador > 0)
{
    // armazenando casa decimal atual na posicao i do vetor
    casasdecimais[i] = divisor / multiplicador;

    divisor %= multiplicador;
    multiplicador /= 10;
    i++;
}

// DAQUI PRA BAIXO ESTAMOS RESOLVENDO O PROBLEMA DO PALINDROMO

// construindo o inverso do vetor "casasdecimais"
inverso = new int[numcasasdecimais];

j = 0;
for (i = numcasasdecimais-1; i >= 0; i--)
{
    inverso [j] = casasdecimais [i];

    j++;
}

palindromo = true;
i = 0;

{
    if (inverso [i] == casasdecimais [i])
        i++;
    else
        palindromo = false;

if (palindromo)
    Console.WriteLine("O número {0} é palíndromo.", numero);
else
    Console.WriteLine("O número {0} não é palíndromo.", numero);

4) Imprimir todos os números palíndromos entre 1 e 1000. 

int numero;
int[] casasdecimais; // numero representado em casas decimais
int[] inverso; // inverso do número que o usuário digita
int numcasasdecimais; // numero de casas decimais do número digitado pelo usuário
int divisor;
int resto;

int i;
int j;

contador = 0;

    // calculando multiplicador
    multiplicador = (int)Math.Pow(10, numcasasdecimais - 1);

    divisor = numero;
    i = 0;
    while (multiplicador > 0)
    {
        // armazenando casa decimal atual na posicao i do vetor
        casasdecimais [i] = divisor / multiplicador;

        divisor %= multiplicador;
        multiplicador /= 10;
        i++;
    }

    // DAQUI PRA BAIXO ESTAMOS RESOLVENDO O PROBLEMA DO PALINDROMO

    // construindo o inverso do vetor "casasdecimais"
    inverso = new int[numcasasdecimais];

    j = 0;
    for (i = numcasasdecimais-1; i >= 0; i--)
    {
        inverso [j] = casasdecimais [i];

    // verificando se o número é palíndromo

        else

    if (palindromo)
    {
        Console.WriteLine(numero);
        contador++;
    }
}

Console.WriteLine("O número de palíndromos entre 1 e 1000 é {0}", contador);

Console.ReadKey();

5) Tentando descobrir se um dado era viciado, um dono de cassino honesto (ha! ha! ha! ha!) o lançou N vezes. Dados os N resultados dos lançamentos, determinar o número de ocorrências de cada face. 

int n; // numero de vezes que o dado será lançado
int lancamento; // resultado do lançamento atual
int[] ocorrencias;

// solicitando ao usuário o número de vezes
Console.Write("Quantas vezes o dado será lançado? ");
n = Int32.Parse(Console.ReadLine());

// alocando vetor de ocorrências
ocorrencias = new int[6];

for (int i = 1; i <= n; i++)
{
    // solicitando ao usuário o valor da face atual
    Console.Write("Digite a face: ");
    lancamento = Int32.Parse(Console.ReadLine());

    ocorrencias [lancamento - 1]++;
}

Console.WriteLine("O número de vezes que saiu a face 1 foi {0}", ocorrencias[0]);
Console.WriteLine("O número de vezes que saiu a face 2 foi {0}", ocorrencias[1]);

Console.WriteLine("O número de vezes que saiu a face 5 foi {0}", ocorrencias[4]);
